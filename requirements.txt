# Internets IRC bot — full runtime dependencies.
#
# Install everything the bot needs for full functionality:
#
#     pip install -r requirements.txt
#
# Each package is documented inline below.  Dev/test/CI deps (pytest,
# coverage, bandit, pip-audit, build, etc.) live in pyproject.toml
# under [project.optional-dependencies] dev — install with
# `pip install -e ".[dev]"` when working on the code itself.
#
# Version policy:  lower bounds are the most recent stable release at
# session time (every package known to be CVE-free at this lower
# bound).  No upper bounds — pip always pulls the newest compatible
# release.  CI runs on Linux, macOS, and Windows across Python 3.10 →
# 3.14, so breakage from a future upstream release surfaces in the
# next push.

# Core HTTP client.  >=2.32.3 carries the PoolManager / verify fix
# (CVE-2024-35195) and the proxy credential leak fix (CVE-2023-32681).
requests>=2.32.3

# Async HTTP client — preferred transport for weather provider calls;
# the bot falls back to requests-in-thread when missing.
# >=3.10.11 carries the chunked-encoding parser fix (CVE-2024-52304).
aiohttp>=3.10.11

# Argon2id — recommended admin password hasher (memory-hard, GPU-resistant).
# Required for `password_hash = argon2$...` in config.local.ini [admin].
argon2-cffi>=23.1.0

# bcrypt — alternative admin password hasher.
# >=4.2.0 is the Rust-backed implementation with current wheels for
# every major platform.  If you're on an exotic platform without a wheel
# and without Rust, comment this out and use scrypt (in stdlib hashlib).
bcrypt>=4.2.0

# PyJWT + cryptography — Apple WeatherKit JWT signing (ES256).
# >=2.10.1 carries the JOSE header-confusion fix (CVE-2024-53861).
# cryptography >=44.0.0 removes legacy TLS 1.0/1.1 support and ships
# vendored OpenSSL with current CVE patches.
PyJWT>=2.10.1
cryptography>=44.0.0

# Hardened XML parser used by modules/qdb.py.  Blocks billion-laughs
# DoS in addition to stdlib's XXE protection.
defusedxml>=0.7.1
